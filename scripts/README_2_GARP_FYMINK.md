```
cd stretcher-match

# Make sure there is an `output` folder
mkdir output

# verify data is available for sequences
ls data

# extract GARP and FYMINK from sequences for species

species=Gorilla_gorilla
Rscript ../scripts/garp_fymink_01_extract.R "$species"

```
